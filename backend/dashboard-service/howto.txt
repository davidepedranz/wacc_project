#build docker image ($PATH_TO_DOCKERFILE = . if you are in ./backend/dashboard-service )
$ docker build -t scala2docker $PATH_TO_DOCKERFILE

#init node as a swarm node
$ docker swarm init

#deploy a stack of containers
$ docker stack deploy -c docker-stack.yml $NAME_WEB_SERVICE

#see web service is running
$ docker service ls

#stop web service
$ docker stack rm $NAME_WEB_SERVICE

#stop docker swarm
$ docker swarm leave --force
